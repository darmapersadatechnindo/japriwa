Main Listener with WhatsApp Event (Client)

});

Available Events (Client → Server)

1. start

Start WhatsApp session


socket.emit('start', { sessionId: 'your-session-id' });

2. status


socket.emit('status', { sessionId: 'your-session-id' });

3. group


socket.emit('group', { sessionId: 'your-session-id' });

4. getContactById


socket.emit('getContactById', {
  sessionId: 'your-session-id',
  contactId: 'contact@c.us'
});

5. info


socket.emit('info', { sessionId: 'your-session-id' });

6. contact


socket.emit('contact', { sessionId: 'your-session-id' });

7. chatList


socket.emit('chatList', { sessionId: 'your-session-id' });

8. profilPic


socket.emit('profilPic', {
  sessionId: 'your-session-id',
  contactId: 'contact@c.us'
});

9. logout


socket.emit('logout', { sessionId: 'your-session-id' });

10. sendMessage


socket.emit('sendMessage', {
  sessionId: 'your-session-id',
  to: '08xxxxxxxx',
  message: 'Hello from Socket.IO'
});

11. sendMedia


socket.emit('sendMedia', {
  sessionId: 'your-session-id',
  to: '08xxxxxxxx',
  fileUrl: 'https://yourdomain.com/file.jpg',
  caption: 'Check this out!'
});

12. getEvents


socket.emit('getEvents', { sessionId: 'your-session-id' });

13. whatsapp:getQr


socket.emit('whatsapp:getQr', { sessionId: 'your-session-id' });

14. showMessage


socket.emit('showMessage', {
  sessionId: 'your-session-id',
  chatId: 'contact@c.us'
});

15. downloadMedia


socket.emit('downloadMedia', {
  sessionId: 'your-session-id',
